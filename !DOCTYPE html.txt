<script>
const lamp = document.getElementById("lamp");
const glow = document.getElementById("glow");

const sentences = [
  "mommy, you make everything calm",
  "i feel safe with you, mommy",
  "rest now, mommy, i‚Äôm here",
  "mommy, you‚Äôre my comfort",
  "my heart feels warm with you, mommy",
  "everything is soft when i think of you, mommy",
  "mommy, i love how you care for me",
  "stay close, mommy, i feel happy",
  "mommy, you make life gentle and sweet",
  "i‚Äôm so lucky to have you, mommy",
  "my baby heart smiles with you, mommy",
  "thank you for being my calm, mommy ü§ç"
];

lamp.addEventListener("click", () => {
  lamp.classList.add("on");
  glow.classList.add("on");
  document.body.style.background = "#f4efe6";

  let index = 0;

  const interval = setInterval(() => {
    if(index >= sentences.length){
      clearInterval(interval);
      return;
    }

    const p = document.createElement("div");
    p.className = "particle";
    p.style.bottom = "60px";

    // Random horizontal offset for natural feel
    const randomOffset = Math.random()*40 - 20;
    p.style.transform = `translateX(${randomOffset}px)`;

    // Wrap letters in span for fade in effect
    const sentence = sentences[index];
    p.innerHTML = "";
    for(let i=0;i<sentence.length;i++){
      const span = document.createElement("span");
      span.innerText = sentence[i];
      span.style.animationDelay = (i*0.05) + "s";
      p.appendChild(span);
    }

    // Add a tiny heart occasionally
    if(Math.random() > 0.7){
      const heart = document.createElement("span");
      heart.className = "heart";
      heart.innerText = "‚ù§";
      p.appendChild(heart);
    }

    document.querySelector(".room").appendChild(p);

    p.style.animation = "floatUp 9s ease forwards";

    // Remove after 9s
    setTimeout(()=>p.remove(),9000);

    index++;
  },2000);

  lamp.style.pointerEvents = "none";
});
</script>
